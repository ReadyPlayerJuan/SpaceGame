#version 430

in vec2 position;
in float wobble;

uniform float time;

const float timeScaleX = 2.5;
const float timeScaleY = 3.1;
const float timeOffsetScaleX = 135.3;
const float timeOffsetScaleY = 89.4;

void main(void) {
    vec2 wobbled_position = position + vec2(0.01 * cos(time * timeScaleX + timeOffsetScaleX * gl_VertexID), 0.01 * sin(time * timeScaleY + timeOffsetScaleY * gl_VertexID));
	gl_Position = vec4(wobbled_position, 0.0, 1.0);
}
